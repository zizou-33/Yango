const loadData = async () => {

    const [transRes, benRes] = await Promise.all([

  } finally {
    setLoading(false);
  }
};




🔧 BACKEND
Endpoint 1 : GET /api/transactions

export async function GET(request: NextRequest) {
  const authHeader = request.headers.get('authorization');
  const token = authHeader.replace('Bearer ', '');
  const decoded = jwt.verify(token, process.env.JWT_SECRET) as { id: string };
  
  const { searchParams } = new URL(request.url);
  const page = parseInt(searchParams.get('page') || '1');
  const limit = parseInt(searchParams.get('limit') || '10');
  const offset = (page - 1) * limit;
  
  const where = {
    OR: [{ senderId: decoded.id }, { receiverId: decoded.id }],
  };
  
  const [transactions, total] = await Promise.all([
    prisma.transaction.findMany({
      where,
      include: {
        sender: { select: { firstName: true, lastName: true, userCode: true } },
        receiver: { select: { firstName: true, lastName: true, userCode: true } },
      },
      orderBy: { createdAt: 'desc' },
      skip: offset,
      take: limit,
    }),
    prisma.transaction.count({ where }),
  ]);
  
  const data = transactions.map((t) => {
    const isSender = t.senderId === decoded.id;
    return {
      ...t,
      direction: isSender ? 'sent' : 'received',
      counterpartyName: isSender
        ? `${t.receiver.firstName} ${t.receiver.lastName}`
        : `${t.sender.firstName} ${t.sender.lastName}`,
      counterpartyCode: isSender ? t.receiver.userCode : t.sender.userCode,
      createdAt: t.createdAt.toISOString(),
    };
  });
  
  return NextResponse.json({
    success: true,
    data: { transactions: data, total, page, totalPages: Math.ceil(total / limit) },
  });
}



Endpoint 2 : GET /api/beneficiaries
export async function GET(request: NextRequest) {
  const authHeader = request.headers.get('authorization');
  const token = authHeader.replace('Bearer ', '');
  const decoded = jwt.verify(token, process.env.JWT_SECRET) as { id: string };
  
  const beneficiaries = await prisma.beneficiary.findMany({
    where: { userId: decoded.id },
    include: {
      beneficiary: {
        select: {
          firstName: true, lastName: true, email: true,
          country: true, currency: true, userCode: true,
        },
      },
    },
    orderBy: { createdAt: 'desc' },
  });
  
  const data = beneficiaries.map((b) => ({
    ...b,
    firstName: b.beneficiary.firstName,
    lastName: b.beneficiary.lastName,
    email: b.beneficiary.email,
    country: b.beneficiary.country,
    currency: b.beneficiary.currency,
    userCode: b.beneficiary.userCode,
  }));
  
  return NextResponse.json({ success: true, data });
}



🗄️ BASE DE DONNÉES
Tables utilisées
Transaction

model Transaction {
  id               String   @id @default(cuid())
  senderId         String
  receiverId       String
  amount           Float
  currency         String
  receivedAmount   Float
  receivedCurrency String
  exchangeRate     Float
  fees             Float    @default(0)
  status           String   @default("completed")
  type             String   @default("send")
  description      String?
  createdAt        DateTime @default(now())
  completedAt      DateTime?
  
  sender   User @relation("Sender", fields: [senderId], references: [id])
  receiver User @relation("Receiver", fields: [receiverId], references: [id])
}


Beneficiary
model Beneficiary {
  id            String   @id @default(cuid())
  userId        String
  beneficiaryId String
  status        String   @default("pending")
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt
  
  user        User @relation("UserBeneficiaries", fields: [userId], references: [id], onDelete: Cascade)
  beneficiary User @relation("AddedByUsers", fields: [beneficiaryId], references: [id], onDelete: Cascade)
  
  @@unique([userId, beneficiaryId])
}

Requêtes exécutées
1. Récupérer les 5 dernières transactions
SELECT * FROM transactions
WHERE sender_id = 'user_id' OR receiver_id = 'user_id'
ORDER BY created_at DESC
LIMIT 5


2. Récupérer les bénéficiaires acceptés
SELECT b.*, u.first_name, u.last_name, u.email, u.country, u.currency, u.user_code
FROM beneficiaries b
JOIN users u ON b.beneficiary_id = u.id
WHERE b.user_id = 'user_id' AND b.status = 'accepted'
ORDER BY b.created_at DESC

PERFORMANCES
Optimisations
Promise.all : Chargement parallèle des transactions et bénéficiaires
Limit 5 : Seulement 5 transactions chargées (pas toutes)
Filter côté serveur : Bénéficiaires filtrés par status='accepted'
useState : State local pour éviter les re-renders inutiles
