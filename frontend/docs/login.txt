BACKEND (app/api/auth/login/route.ts)
Endpoint
Method : POST
URL : /api/auth/login
Body : { email: string, password: string }

export async function POST(request: NextRequest) {
  // 1. Récupérer les données du body
  const { email, password } = await request.json();
  
  // 2. Chercher l'utilisateur dans la base de données
  const user = await prisma.user.findUnique({ where: { email } });
  
  // 3. Vérifier si l'utilisateur existe
  if (!user) {
    return NextResponse.json({ error: 'Identifiants invalides' }, { status: 401 });
  }
  
  // 4. Vérifier le mot de passe avec bcrypt
  const isMatch = await bcrypt.compare(password, user.password);
  if (!isMatch) {
    return NextResponse.json({ error: 'Identifiants invalides' }, { status: 401 });
  }
  
  // 5. Générer un token JWT
  const token = jwt.sign(
    { id: user.id, email: user.email },
    process.env.JWT_SECRET,
    { expiresIn: '7d' }
  );
  
  // 6. Calculer les statistiques de l'utilisateur
  const totalSent = await prisma.transaction.aggregate({
    where: { senderId: user.id, status: 'completed' },
    _sum: { amount: true }
  });
  
  // 7. Retourner les données utilisateur + token
  return NextResponse.json({
    success: true,
    data: {
      user: { ...user, totalSent, totalReceived, sentCount, receivedCount },
      token
    }
  });
}

Sécurité
Mot de passe hashé avec bcrypt (10 rounds)
Token JWT avec expiration de 7 jours
Validation des entrées côté serveur


 BASE DE DONNÉES (prisma/schema.prisma)
Table User (utilisée pour la connexion)
model User {
  id        String   @id @default(cuid())
  email     String   @unique          // Utilisé pour la connexion
  phone     String?
  password  String                    // Hashé avec bcrypt
  firstName String
  lastName  String
  country   String
  birthDate String
  currency  String   @default("XAF")
  userCode  String   @unique          // Code unique pour les bénéficiaires
  balance   Float    @default(327980) // 500 EUR convertis
  isVerified Boolean @default(false)
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
  
  // Relations
  sentTransactions     Transaction[]
  receivedTransactions Transaction[]
  beneficiaries        Beneficiary[]
  addedBy              Beneficiary[]
  notifications        Notification[]
}


Champs importants pour le login
email : Identifiant unique pour la connexion
password : Mot de passe hashé (jamais en clair)
id : Identifiant unique utilisé dans le token JWT
