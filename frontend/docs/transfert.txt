4. Calcul automatique du montant reçu


const calculateReceivedAmount = (): number => {
  if (!selectedBeneficiary || !amount) return 0;
  const rate = exchangeRates[`XAF_${selectedBeneficiary.currency}`] || 0.001524;
  return parseFloat(amount) * rate;
};

const calculateFees = (): number => {
  return parseFloat(amount || '0') * 0.005; // 0.5% de frais
};

Logique de transfert

const handleConfirmTransfer = async () => {
  if (pinCode.length !== 4) {
    toast.error('Le code PIN doit contenir 4 chiffres');
    return;
  }

  setLoading(true);
  try {
    const receivedAmount = calculateReceivedAmount();
    const fees = calculateFees();
    const rate = exchangeRates[`XAF_${selectedBeneficiary!.currency}`] || 0.001524;

    await transactionApi.create({
      receiverId: selectedBeneficiary!.beneficiaryId,
      amount: parseFloat(amount),
      currency: 'XAF',
      receivedAmount,
      receivedCurrency: selectedBeneficiary!.currency,
      exchangeRate: 1 / rate,
      fees,
      type: 'send',
      description: `Transfert à ${selectedBeneficiary!.firstName} ${selectedBeneficiary!.lastName}`,
      pinCode,
    });

    toast.success('Transfert effectué avec succès!');
    router.push('/history');
  } catch (error) {
    toast.error('Erreur lors du transfert');
  } finally {
    setLoading(false);
    setShowPinModal(false);
    setPinCode('');
  }
};



BACKEND (app/api/transactions/route.ts)
Endpoint POST /api/transactions
Logique serveur

export async function POST(request: NextRequest) {
  // 1. Vérification de l'authentification
  const authHeader = request.headers.get('authorization');
  const token = authHeader.replace('Bearer ', '');
  const decoded = jwt.verify(token, process.env.JWT_SECRET) as { id: string };
  
  // 2. Récupération des données
  const body = await request.json();
  const { receiverId, amount, currency, receivedAmount, receivedCurrency, exchangeRate, fees, type, description, pinCode } = body;
  
  // 3. Validation du montant
  if (!amount || amount <= 0) {
    return NextResponse.json({ error: 'Le montant doit être supérieur à 0' }, { status: 400 });
  }
  
  if (amount > 10000000) {
    return NextResponse.json({ error: 'Le montant maximum est de 10 000 000' }, { status: 400 });
  }
  
  // 4. Récupération de l'expéditeur
  const sender = await prisma.user.findUnique({ where: { id: decoded.id } });
  if (!sender) {
    return NextResponse.json({ error: 'Expéditeur non trouvé' }, { status: 404 });
  }
  
  // 5. Vérification du code PIN
  if (sender.pinCode !== pinCode) {
    return NextResponse.json({ error: 'Code PIN incorrect' }, { status: 400 });
  }
  
  // 6. Vérification du solde
  if (sender.balance < amount + fees) {
    return NextResponse.json({ 
      error: `Solde insuffisant. Solde actuel: ${sender.balance} ${sender.currency}` 
    }, { status: 400 });
  }
  
  // 7. Récupération du destinataire
  const receiver = await prisma.user.findUnique({ where: { id: receiverId } });
  if (!receiver) {
    return NextResponse.json({ error: 'Destinataire non trouvé' }, { status: 404 });
  }
  
  // 8. Calcul des nouveaux soldes
  const newSenderBalance = sender.balance - (amount + fees);
  const newReceiverBalance = receiver.balance + receivedAmount;
  
  // 9. Transaction atomique (tout ou rien)
  const transaction = await prisma.$transaction(async (tx) => {
    // a. Créer la transaction
    const newTransaction = await tx.transaction.create({
      data: {
        senderId: decoded.id,
        receiverId,
        amount: parseFloat(amount.toString()),
        currency,
        receivedAmount: parseFloat(receivedAmount.toString()),
        receivedCurrency,
        exchangeRate: parseFloat(exchangeRate.toString()),
        fees: parseFloat(fees.toString()),
        status: 'completed',
        type: type || 'send',
        description: description || '',
        completedAt: new Date(),
      },
    });
    
    // b. Débiter l'expéditeur
    await tx.user.update({
      where: { id: decoded.id },
      data: { balance: newSenderBalance },
    });
    
    // c. Créditer le destinataire
    await tx.user.update({
      where: { id: receiverId },
      data: { balance: newReceiverBalance },
    });
    
    // d. Créer les notifications
    await tx.notification.create({
      data: {
        userId: receiverId,
        type: 'transfer_received',
        message: `Vous avez reçu ${receivedAmount} ${receivedCurrency} de ${sender.firstName} ${sender.lastName}`,
      },
    });
    
    await tx.notification.create({
      data: {
        userId: decoded.id,
        type: 'transfer_sent',
        message: `Vous avez envoyé ${amount} ${currency} à ${receiver.firstName} ${receiver.lastName}`,
      },
    });
    
    return newTransaction;
  });
  
  // 10. Retourner la confirmation
  return NextResponse.json({
    success: true,
    message: 'Transaction completed',
    data: { ...transaction, createdAt: transaction.createdAt.toISOString() },
  }, { status: 201 });
}

🔐 SÉCURITÉ
Code PIN
Vérifié côté serveur avant chaque transaction
4 chiffres uniquement
Stocké en clair dans la base de données (pour validation rapide)
Validation des montants
Montant > 0
Montant < 10 000 000 (limite maximale)
Solde suffisant (amount + fees)
Transaction atomique
Utilisation de prisma.$transaction()
Si une étape échoue, tout est annulé
Garantie d'intégrité des données
Authentification
Token JWT requis pour chaque requête
Vérification de l'identité de l'expéditeur
Impossible de transférer à soi-même (vérifié côté bénéficiaires)
🐛 GESTION DES ERREURS
Erreurs frontend
Aucun bénéficiaire sélectionné → "Veuillez remplir tous les champs"
Montant <= 0 → "Le montant doit être supérieur à 0"
Solde insuffisant → "Solde insuffisant"
PIN != 4 chiffres → "Le code PIN doit contenir 4 chiffres"
Erreurs backend
Token invalide → 401 Unauthorized
Montant invalide → 400 Bad Request
PIN incorrect → 400 Bad Request
Solde insuffisant → 400 Bad Request
Destinataire non trouvé → 404 Not Found
Erreur serveur → 500 Internal Server Error
